def get_sum(data):
    result = 0
    for datum in data:
        result = result + datum
    return result

def get_len(data):
    result = 0
    for datum in data:
        result = result + 1
    return result

def get_average(data):
    total = get_sum(data)
    n = get_len(data)
    result = total / n
    return result


score = [50, 60, 70]
average = get_average(score)
print(average)

import math
candidates = {
  'alan': [8, 14, 6, 8, 14, 9, 14, 9, 15, 5],
  'brad': [11, 4, 11, 7, 9, 7, 8, 7, 6],
  'cate': [16, 22, 15, 12, 3, 20, 17, 13, 23],
  'dave': [24, 15, 18, 18, 12, 9, 19, 23, 13, 14, 18],
}

sd = []

for name, scores in candidates.items():
    # 합계 구하기
    total = sum(scores)
    # value의 전체 개수 구하기
    n = len(scores)
    # 평균구하기
    mean = total / n

    # 분산구하기
    vsum = 0
    for x in scores:
      vsum = vsum + (x - mean)**2  # 
    var = vsum / n-1

# 표준편차 구하기
    std = math.sqrt(var)
    sd.append(std)


named_sd = list(zip(candidates.keys(), sd))

print(named_sd)
